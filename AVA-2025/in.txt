declare function printGreeting() {
	display("Hello");
	declare symbol message: "Test program";
	display(message);
}

declare function workWithSymbols() {
	declare symbol ch1: 'A';
	declare symbol ch2: 'B';
	
	display(ch1);
	display(ch2);
}

declare function stringComparison() {
	declare symbol str1: "Hello";
	declare symbol str2: "World";
	
	declare int cmpResult;
	cmpResult: StrCmp(str1, str2);
	display(cmpResult);
}

declare function loopExample() {
	declare int counter;
	counter: 0;
	
	for (counter: 5) {
		display(counter);
		counter++;
	}
}

declare int function testUnaryOperations() {
	declare int num;
	num: 10;
	num++;
	display(num);
	num--;
	display(num);
	~num;
	display(num);
	return num;
}

main {
	printGreeting();
	workWithSymbols();
	stringComparison();
	
	declare int num1;
	declare int num2;
	
	num1: 15;
	num2: 25;
	
	loopExample();
	
	testUnaryOperations();
}

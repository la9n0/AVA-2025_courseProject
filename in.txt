number function factorial(number n)
[
    var number result;
    result = 1;
    cycle n [
        result = result * n;
        dec n;
    ]
    return result;
]

number function power(number base, number exp)
[
    var number res;
    res = 1;
    cycle exp [
        res = res * base;
        dec exp;
    ]
    return res;
]

number function calculate(number a, number b)
[
    var number sum;
    var number diff;
    var number prod;
    var number quot;
    var number mod;
    var number result;
    sum = a + b;
    diff = a - b;
    prod = a * b;
    quot = a / b;
    mod = a % b;
    
    result = sum + diff + prod + quot + mod;
    return result;
]
